    "${CMAKE_CURRENT_SOURCE_DIR}/*.cpp"
)

file(GLOB_RECURSE RESOURCE_HEADERS 
    "${CMAKE_CURRENT_SOURCE_DIR}/*.h"
    "${CMAKE_CURRENT_SOURCE_DIR}/*.hpp"
)

target_sources(${ENGINE_NAME} PRIVATE ${RESOURCE_SOURCES} ${RESOURCE_HEADERS})
